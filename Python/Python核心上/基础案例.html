str_info = """**************************************************
欢迎使用【学生信息管理系统】V1.0
请选择你想要进行的操作
1. 新建学生信息
2. 显示全部信息
3. 查询学生信息
4. 修改学生信息
5. 删除学生信息

0. 退出系统
**************************************************"""

# 在自己不知道要去运行多少次的前提下  死循环搭配 break 去使用

    # 1.程序启动，显示信息管理系统欢迎界面，并显示功能菜单print打印

    # 2.用户用数字选择不同的功能

        print('0. 退出系统')
        break # 退出死循环
    # 处理除了以上的其他情况 保证代码严谨性
    else:
        print('请输入正确的选项:')

students = [
    {'name': '丸子', 'chinese': 86, 'math': 86, 'english': 86, 'total': 258},
    {'name': '山禾', 'chinese': 86, 'math': 86, 'english': 86, 'total': 258},

    if stu['name'] == del_name: # 找到了

完整的学生管理系统 ​

python
# 1. 程序启动，显示信息管理系统欢迎界面，并显示功能菜单 print打印
# 2. 用户用数字选择不同的功能 input
# 3. 根据功能选择，执行不同的功能 if elif else 搭配
# 4. 需要记录学生的 **姓名**、**语文成绩**、**数学成绩**、**英语成绩** 、**总分** 数据容器的使用
# 5. 如果查询到指定的学生信息，用户可以选择 **修改** 或者 **删除** 信息  考验的是数据容器增删改查
# 6. 进入或退出时加载或保存数据  with open

str_info = """**************************************************
欢迎使用【学生信息管理系统】V1.0
请选择你想要进行的操作
1. 新建学生信息
2. 显示全部信息
3. 查询学生信息
4. 修改学生信息
5. 删除学生信息

0. 退出系统
**************************************************"""
import json
with open('students.json',mode='r',encoding='utf-8') as f:
    student_s = f.read()
    # print(student_s)
    # print(type(student_s))
# # 因为我们执行对学生信息的增删改查的操作都要基于列表的数据类型 str是没有这些方法的
# 就需要将str转化为列表类型
# json.loads
student = json.loads(student_s)

# 在自己不知道要去运行多少次的前提下  死循环搭配 break 去使用
while True:
    # 1.程序启动，显示信息管理系统欢迎界面，并显示功能菜单print打印
    print(str_info)
    # 2.用户用数字选择不同的功能
    action = input('请输入你的需求序号:') # str
    # 3.根据功能选择，执行不同的功能
    if action == '1': # 数据进行等式比较的(数据类型是要一致)
        print('1. 新建学生信息')
        name = input('请输入你的名字:')
        chinese = int(input('请输入你的语文成绩:'))
        math = int(input('请输入你的数学成绩:'))
        english = int(input('请输入你的英语成绩:'))
        # 总分的统计
        total = chinese + math + english
        # 定义保存数据的容器
        stu = {'name': name, 'chinese': chinese, 'math': math, 'english': english, 'total': total}
        # 将字典数据添加到列表里面去
        student.append(stu)
        print('##### ----- 添加成功 ----- #####')


    elif action == '2':
        print('2. 显示全部信息')
        print('姓名\t\t语文\t数学\t\t英语\t总分')
        for stu in student:
            # print(stu) # stu 列表里面遍历出来的一个一个字典
            # 只能是单包双 或者双包单 不能是单包单 或者双包双
            print(f"{stu['name']}\t\t{stu['chinese']}\t\t{stu['math']}\t\t{stu['english']}\t\t{stu['total']}")
        print('##### ----- 显示成功 ----- #####')
    elif action == '3':
        print('3. 查询学生信息')
        # 输入名字用于查找
        search_name = input('请输入你的名字:')

        for stu in student:
            # 如果遍历的学生字典中名字是要查找的学生姓名
            if stu['name'] == search_name:  # 找到了
                # 展示的内容
                print('姓名\t语文\t数学\t英语\t总分')
                print(f"{stu['name']}\t{stu['chinese']}\t\t{stu['math']}\t\t{stu['english']}\t\t{stu['total']}")
                break  # 找到了后面的for循环就没必要继续执行了 break退出了离他最近的for循环
                # 如果break执行了 就表示他是一个非正常跳出  else是不会被执行的
        else:  # 当循环正常结束的时候 else会被执行
            print('该学生不存在, 请检查名字是否输入正确!')

    elif action == '4':
        print('4. 修改学生信息')
        # 输入学生的姓名, 用于修改
        modify_name = input('请输入你要修改的名字:')
        for stu in student:
            if stu['name'] == modify_name:  # 找到了
                # 方法二
                print('不想修改改内容,请敲回车跳过')  # 此时敲的回车就相当于一个空字符
                name = input('请重新输入你的名字:')
                chinese = input('请重新输入你的语文成绩:')
                math = input('请重新输入你的数学成绩:')
                english = input('请重新输入你的英语成绩:')
                if name:  # 非空即是true # 空字符串的布尔判断结果为False 使用自己原来的名字
                    stu['name'] = name
                if chinese:
                    stu['chinese'] = int(chinese)
                if math:
                    stu['math'] = int(math)
                if english:
                    stu['english'] = int(english)
                # 总分应该是前面的数据重新覆盖更新之后再去计算
                stu['total'] = stu['chinese'] + stu['math'] + stu['english']

                # 方法一
                # name = input('请重新输入你的名字:')
                # chinese = int(input('请重新输入你语文成绩:'))
                # math = int(input('请重新输入你的数学成绩:'))
                # english = int(input('请重新输入你的英语成绩'))
                # total = chinese + math + english # 总分
                # # 通过字典键覆盖去达到修改的目的
                # stu['name'] = name
                # stu['chinese'] = chinese
                # stu['math'] = math
                # stu['english'] = english
                # stu['total'] = total
                break

        else:  # 没有找到 没有出break 就会去执行else的语句
            print('该学生不存在, 请检查名字是否输入正确!')

    elif action == '5':
        print('5. 删除学生信息')
        # 输入要查找的名字 用于删除
        del_name = input('请输入你要删除的名字:')
        for stu in student:
            if stu['name'] == del_name:  # 找到了
                # pop(默认是删除最后一个列表数据, 也可以指定索引在列表中删除数据)
                # index students.index(stu)获取数据在列表里面的索引
                # del 列表的名字 才会直接抹除列表的定义

                # del students[students.index(stu)]
                # del students
                # students.pop(students.index(stu))
                # remove(根据数据在列表中删除)
                student.remove(stu)
                break

        else:
            print('该学生不存在, 请检查名字是否输入正确!')


    elif action == '0':
        print('0. 退出系统')
        # 当退出系统的时候需要把数据保存到文件里面去
        with open('students.json', mode='w', encoding='utf-8') as f:
            # write只能写入 二进制和str
            # 在保存的时候需要列表的类型转回str类型
            # f.write(str(students)) # 直接通过str强转的数据是不行 会影响数据的取值
            # json.dumps 将列表类型序列化str类型
            # 这个是因为json在进行序列化时，默认使用的是编码SCAII，而中文为Unicode编码，ASCII中不包含中文，所以出现了乱码。
            student_str = json.dumps(student, ensure_ascii=False)
            # print(student_str)
            # print(type(student_str))
            f.write(student_str)
        break # 退出死循环
    # 处理除了以上的其他情况 保证代码严谨性
    else:
        print('请输入正确的选项:')

python
# 1. 程序启动，显示信息管理系统欢迎界面，并显示功能菜单 print打印
# 2. 用户用数字选择不同的功能 input
# 3. 根据功能选择，执行不同的功能 if elif else 搭配
# 4. 需要记录学生的 **姓名**、**语文成绩**、**数学成绩**、**英语成绩** 、**总分** 数据容器的使用
# 5. 如果查询到指定的学生信息，用户可以选择 **修改** 或者 **删除** 信息  考验的是数据容器增删改查
# 6. 进入或退出时加载或保存数据  with open
# 封装一个用于重新赋值的函数逻辑
def insert_student():
    """信息的录入"""
    name = input('请重新输入你的名字:')
    chinese = input('请重新输入你的语文成绩:')
    math = input('请重新输入你的数学成绩:')
    english = input('请重新输入你的英语成绩:')
    stu = {}     # 现在是单独封装函数 stu是需要被重新定定义的
    if name:  # 非空即是true # 空字符串的布尔判断结果为False 使用自己原来的名字
        stu['name'] = name
    if chinese:
        stu['chinese'] = int(chinese)
    if math:
        stu['math'] = int(math)  # stu是一个空   如果不赋值是不会这个键的 这个问题 我们后面讲类与对象会去解决
    if english:
        stu['english'] = int(english)
    # 总分应该是前面的数据重新覆盖更新之后再去计算
    stu['total'] = stu['chinese'] + stu['math'] + stu['english']
    return stu # stu就是保存着学生信息的字典 return 是实现函数内外数据共享 在任何地方去调用这个函数 stu

# 查询  修改  删除 都需要找学生
def search_student(stu_name,student_s):# stu_name是要找的名字,student_s在哪里找范围
    for stu in student_s:
        if stu_name == stu['name']: # 找到了
            return stu  # stu 学生所在的整个字典数据

        # else不写默认返回的是空(None)
str_info = """**************************************************
欢迎使用【学生信息管理系统】V1.0
请选择你想要进行的操作
1. 新建学生信息
2. 显示全部信息
3. 查询学生信息
4. 修改学生信息
5. 删除学生信息

0. 退出系统
**************************************************"""
import json
with open('students.json',mode='r',encoding='utf-8') as f:
    student_s = f.read()
    # print(student_s)
    # print(type(student_s))
# # 因为我们执行对学生信息的增删改查的操作都要基于列表的数据类型 str是没有这些方法的
# 就需要将str转化为列表类型
# json.loads
student = json.loads(student_s)

# 在自己不知道要去运行多少次的前提下  死循环搭配 break 去使用
while True:
    # 1.程序启动，显示信息管理系统欢迎界面，并显示功能菜单print打印
    print(str_info)
    # 2.用户用数字选择不同的功能
    action = input('请输入你的需求序号:') # str
    # 3.根据功能选择，执行不同的功能
    if action == '1': # 数据进行等式比较的(数据类型是要一致)
        print('1. 新建学生信息')
        stu = insert_student()   # stu就是函数返回的字典
        # 将字典数据添加到列表里面去
        student.append(stu)
        print('##### ----- 添加成功 ----- #####')


    elif action == '2':
        print('2. 显示全部信息')
        print('姓名\t语文\t数学\t英语\t总分')
        for stu in student:
            # print(stu) # stu 列表里面遍历出来的一个一个字典
            # 只能是单包双 或者双包单 不能是单包单 或者双包双
            print(f"{stu['name']}\t{stu['chinese']}\t\t{stu['math']}\t\t{stu['english']}\t\t{stu['total']}")
        print('##### ----- 显示成功 ----- #####')
    elif action == '3':
        print('3. 查询学生信息')
        # 输入名字用于查找
        search_name = input('请输入你的名字:')

        stu = search_student(search_name,student)
        # 利用非空即是True
        if stu:
                # 展示的内容
                print('姓名\t\t语文\t数学\t\t英语\t总分')
                print(f"{stu['name']}\t\t{stu['chinese']}\t\t{stu['math']}\t\t{stu['english']}\t\t{stu['total']}")

        else:  # 当循环正常结束的时候 else会被执行
            print('该学生不存在, 请检查名字是否输入正确!')

    elif action == '4':
        print('4. 修改学生信息')
        # 输入学生的姓名, 用于修改
        modify_name = input('请输入你要修改的名字:')
        stu = search_student(modify_name,student)
        if stu: #如果找到返回的是一个字典 如果没有找到返回的是空字符
                new_stu = insert_student() # 返回的是一个保存着学生数据的字典
                stu.update(new_stu) # 利用 相同键覆盖不同键合并的特性去修改字典
        else:  # 没有找到 没有出break 就会去执行else的语句
            print('该学生不存在, 请检查名字是否输入正确!')

    elif action == '5':
        print('5. 删除学生信息')
        # 输入要查找的名字 用于删除
        del_name = input('请输入你要删除的名字:')
        stu = search_student(del_name,student)
        if stu:  # 找到了
                # pop(默认是删除最后一个列表数据, 也可以指定索引在列表中删除数据)
                # index students.index(stu)获取数据在列表里面的索引
                # del 列表的名字 才会直接抹除列表的定义
                # del students[students.index(stu)]
                # del students
                # students.pop(students.index(stu))
                # remove(根据数据在列表中删除)
                student.remove(stu)

        else:
            print('该学生不存在, 请检查名字是否输入正确!')


    elif action == '0':
        print('0. 退出系统')
        # 当退出系统的时候需要把数据保存到文件里面去
        with open('students.json', mode='w', encoding='utf-8') as f:
            # write只能写入 二进制和str
            # 在保存的时候需要列表的类型转回str类型
            # f.write(str(students)) # 直接通过str强转的数据是不行 会影响数据的取值
            # json.dumps 将列表类型序列化str类型
            # 这个是因为json在进行序列化时，默认使用的是编码SCAII，而中文为Unicode编码，ASCII中不包含中文，所以出现了乱码。
            student_str = json.dumps(student, ensure_ascii=False)
            # print(student_str)
            # print(type(student_str))
            f.write(student_str)
        break # 退出死循环
    # 处理除了以上的其他情况 保证代码严谨性
    else:
        print('请输入正确的选项:')

def max_year(year):
    # % 取余 余数为零 那么就表可以被整除

import copy

    'cate': '童书馆',
    'sub_cate': None # None 相当于空的盒子
}
# 定义了一个列表 列表里面存放着5条数据
sub_cate = ['科普百科', '儿童文学', '幼儿启蒙', '动漫卡通', '少儿英语']

# 定义一个空的列表
all_cate = []

# [{'cate': '童书馆','sub_cate': '儿童文学'},{'cate': '童书馆','sub_cate': '儿童文学'}...]
for cate in sub_cate:
    print(cate)

    # 对字典的键进行重新的赋值操作
    data['sub_cate'] = cate
    # 在这里每一次的字典id是一样的 就说明字典引用的是同一个对象(自始至终都是在使用一个字典)
    #在字典里面键一样会进行数据的覆盖
    print(id(data))
    # 将字典添加到列表里面去
    # 浅拷贝 只适用于一维维度
    all_cate.append(copy.copy(data))  # 在添加的时候 由于是同一个字典 键是一样 先跟新前面的内容然后覆盖 再去添加

    all_cate.append(copy.copy(data))  # 在添加的时候 由于是同一个字典 键是一样 先跟新前面的内容然后覆盖 再去添加
